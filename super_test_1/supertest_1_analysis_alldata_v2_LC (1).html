# For small sample size adjustment
# Adjust the sample size, N
def adjust_n(N):
    return N+(1.96**2)

# Adjust the proportion of successes, p
def adjust_p(p, N):
    return (p+1.92)/(N+3.84)

T2_p = results['Task 2 (N)']['success']
T8_p = results['Task 8 (N)']['success']

# To calculate the confidence intervals the formula used with these values is:
#    CI_95 = 1.96 * SQRT((adjusted_p * (1 - adjusted_p))/adjusted_n)
adjust_n(N), adjust_p(T2_p,N), adjust_p(T8_p,N)

T2_CI = 1.96 * np.sqrt((adjust_p(T2_p,N) * (1 - adjust_p(T2_p,N)))/adjust_n(N))
T8_CI = 1.96 * np.sqrt((adjust_p(T2_p,N) * (1 - adjust_p(T8_p,N)))/adjust_n(N))

### For plotting graphs
def grapher_errorbar(df, cols, title, cis, labels):
    plot_df = df[cols]
    colours=['#9d9d9c','#074f5f','#42bac7']
    ax = plot_df.plot(kind='bar', legend=False, color=colours, edgecolor = "none", rot = 0, yerr=cis)
    ax.set_xticklabels(list(plot_df[cols].index))
    ax.set_ylim([0,1])
    ax.yaxis.set_major_formatter(FuncFormatter(lambda y, _: '{:.0%}'.format(y)))
    ax.set_title(title)
    if not(not labels):
        ax.legend(labels,loc='best')
    plt.savefig("super-1-graphs/"+title+".svg")

def grapher_stacked(df, cols, title, labels, cis):
    plot_df = df[cols].unstack(level=0).sort_index(ascending=False)
    colours=['#42bac7','#074f5f','#9d9d9c']
    ax = plot_df.plot(kind='bar', legend=True, color=colours, edgecolor = "none", rot = 0, yerr=cis)
    ax.set_xticklabels(list(plot_df.index))
    ax.set_ylim([0,1])
    ax.yaxis.set_major_formatter(FuncFormatter(lambda y, _: '{:.0%}'.format(y)))
    ax.set_title(title)
    ax.legend(labels,loc='best')
    plt.savefig("super-1-graphs/"+title+".svg")

### For plotting graphs with stacked data sets
def grapher_stacked_N(df, cols, title):
    plot_df = df[cols].unstack(level=1)
    #plot_df = task2_results[['Task 2 (%)']]
    colours=['#42bac7','#074f5f','#9d9d9c']
    ax = plot_df.plot(kind='bar', legend=True, color=colours, edgecolor = "none", rot = 0)
    ax.set_xticklabels(list(plot_df[cols].index))
    ax.set_title(title)
    plt.savefig("super-1-graphs/"+title+".svg")
